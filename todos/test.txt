import json
import logging
import os
import time
import uuid
import boto3


dynamodb = boto3.resource('dynamodb')

def put_item(data, timestamp):
    table = dynamodb.Table(os.environ['DYNAMODB_TABLE'])
    
    item = {
        'id': str(uuid.uuid1()),
        'text': data['text'],
        'checked': False,
        'createdAt': timestamp,
        'updatedAt': timestamp,
    }

    # write the todo to the database
    table.put_item(Item=item)
    return item

put_item()
